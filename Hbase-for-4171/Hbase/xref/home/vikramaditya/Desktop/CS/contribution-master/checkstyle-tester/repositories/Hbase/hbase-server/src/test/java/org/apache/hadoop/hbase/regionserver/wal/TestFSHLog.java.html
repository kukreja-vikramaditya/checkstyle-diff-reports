159   protected void addEdits(WAL log,
160                           HRegionInfo hri,
161                           HTableDescriptor htd,
162                           int times,
163                           MultiVersionConcurrencyControl mvcc)
164       throws IOException {
165     final byte[] row = Bytes.toBytes("row");
166     for (int i = 0; i < times; i++) {
167       long timestamp = System.currentTimeMillis();
168       WALEdit cols = new WALEdit();
169       cols.add(new KeyValue(row, row, row, timestamp, row));
170       WALKey key = new WALKey(hri.getEncodedNameAsBytes(), htd.getTableName(),
171           WALKey.NO_SEQUENCE_ID, timestamp, WALKey.EMPTY_UUIDS, HConstants.NO_NONCE,
172           HConstants.NO_NONCE, mvcc);
173       log.append(htd, hri, key, cols, true);
174     }
175     log.sync();
176   }

188   /**
189    * tests the log comparator. Ensure that we are not mixing meta logs with non-meta logs (throws
190    * exception if we do). Comparison is based on the timestamp present in the wal name.
191    * @throws Exception
192    */
193   @Test 
194   public void testWALComparator() throws Exception {
195     FSHLog wal1 = null;
196     FSHLog walMeta = null;
197     try {
198       wal1 = new FSHLog(fs, FSUtils.getRootDir(conf), dir.toString(),
199           HConstants.HREGION_OLDLOGDIR_NAME, conf, null, true, null, null);
200       LOG.debug("Log obtained is: " + wal1);
201       Comparator<Path> comp = wal1.LOG_NAME_COMPARATOR;
202       Path p1 = wal1.computeFilename(11);
203       Path p2 = wal1.computeFilename(12);
204       // comparing with itself returns 0
205       assertTrue(comp.compare(p1, p1) == 0);
206       // comparing with different filenum.
207       assertTrue(comp.compare(p1, p2) < 0);
208       walMeta = new FSHLog(fs, FSUtils.getRootDir(conf), dir.toString(),
209           HConstants.HREGION_OLDLOGDIR_NAME, conf, null, true, null,
210           DefaultWALProvider.META_WAL_PROVIDER_ID);
211       Comparator<Path> compMeta = walMeta.LOG_NAME_COMPARATOR;
212 
213       Path p1WithMeta = walMeta.computeFilename(11);
214       Path p2WithMeta = walMeta.computeFilename(12);
215       assertTrue(compMeta.compare(p1WithMeta, p1WithMeta) == 0);
216       assertTrue(compMeta.compare(p1WithMeta, p2WithMeta) < 0);
217       // mixing meta and non-meta logs gives error
218       boolean ex = false;
219       try {
220         comp.compare(p1WithMeta, p2);
221       } catch (IllegalArgumentException e) {
222         ex = true;
223       }
224       assertTrue("Comparator doesn't complain while checking meta log files", ex);
225       boolean exMeta = false;
226       try {
227         compMeta.compare(p1WithMeta, p2);
228       } catch (IllegalArgumentException e) {
229         exMeta = true;
230       }
231       assertTrue("Meta comparator doesn't complain while checking log files", exMeta);
232     } finally {
233       if (wal1 != null) {
234         wal1.close();
235       }
236       if (walMeta != null) {
237         walMeta.close();
238       }
239     }

242   /**
243    * On rolling a wal after reaching the threshold, {@link WAL#rollWriter()} returns the
244    * list of regions which should be flushed in order to archive the oldest wal file.
245    * <p>
246    * This method tests this behavior by inserting edits and rolling the wal enough times to reach
247    * the max number of logs threshold. It checks whether we get the "right regions" for flush on
248    * rolling the wal.
249    * @throws Exception
250    */
251   @Test 
252   public void testFindMemStoresEligibleForFlush() throws Exception {
253     LOG.debug("testFindMemStoresEligibleForFlush");
254     Configuration conf1 = HBaseConfiguration.create(conf);
255     conf1.setInt("hbase.regionserver.maxlogs", 1);
256     FSHLog wal = new FSHLog(fs, FSUtils.getRootDir(conf1), dir.toString(),
257         HConstants.HREGION_OLDLOGDIR_NAME, conf1, null, true, null, null);
258     HTableDescriptor t1 =
259         new HTableDescriptor(TableName.valueOf("t1")).addFamily(new HColumnDescriptor("row"));
260     HTableDescriptor t2 =
261         new HTableDescriptor(TableName.valueOf("t2")).addFamily(new HColumnDescriptor("row"));
262     HRegionInfo hri1 =
263         new HRegionInfo(t1.getTableName(), HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
264     HRegionInfo hri2 =
265         new HRegionInfo(t2.getTableName(), HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
266     // add edits and roll the wal
267     MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
268     try {
269       addEdits(wal, hri1, t1, 2, mvcc);
270       wal.rollWriter();
271       // add some more edits and roll the wal. This would reach the log number threshold
272       addEdits(wal, hri1, t1, 2, mvcc);
273       wal.rollWriter();
274       // with above rollWriter call, the max logs limit is reached.
275       assertTrue(wal.getNumRolledLogFiles() == 2);
276 
277       // get the regions to flush; since there is only one region in the oldest wal, it should
278       // return only one region.
279       byte[][] regionsToFlush = wal.findRegionsToForceFlush();
280       assertEquals(1, regionsToFlush.length);
281       assertEquals(hri1.getEncodedNameAsBytes(), regionsToFlush[0]);
282       // insert edits in second region
283       addEdits(wal, hri2, t2, 2, mvcc);
284       // get the regions to flush, it should still read region1.
285       regionsToFlush = wal.findRegionsToForceFlush();
286       assertEquals(regionsToFlush.length, 1);
287       assertEquals(hri1.getEncodedNameAsBytes(), regionsToFlush[0]);
288       // flush region 1, and roll the wal file. Only last wal which has entries for region1 should
289       // remain.
290       flushRegion(wal, hri1.getEncodedNameAsBytes(), t1.getFamiliesKeys());
291       wal.rollWriter();
292       // only one wal should remain now (that is for the second region).
293       assertEquals(1, wal.getNumRolledLogFiles());
294       // flush the second region
295       flushRegion(wal, hri2.getEncodedNameAsBytes(), t2.getFamiliesKeys());
296       wal.rollWriter(true);
297       // no wal should remain now.
298       assertEquals(0, wal.getNumRolledLogFiles());
299       // add edits both to region 1 and region 2, and roll.
300       addEdits(wal, hri1, t1, 2, mvcc);
301       addEdits(wal, hri2, t2, 2, mvcc);
302       wal.rollWriter();
303       // add edits and roll the writer, to reach the max logs limit.
304       assertEquals(1, wal.getNumRolledLogFiles());
305       addEdits(wal, hri1, t1, 2, mvcc);
306       wal.rollWriter();
307       // it should return two regions to flush, as the oldest wal file has entries
308       // for both regions.
309       regionsToFlush = wal.findRegionsToForceFlush();
310       assertEquals(2, regionsToFlush.length);
311       // flush both regions
312       flushRegion(wal, hri1.getEncodedNameAsBytes(), t1.getFamiliesKeys());
313       flushRegion(wal, hri2.getEncodedNameAsBytes(), t2.getFamiliesKeys());
314       wal.rollWriter(true);
315       assertEquals(0, wal.getNumRolledLogFiles());
316       // Add an edit to region1, and roll the wal.
317       addEdits(wal, hri1, t1, 2, mvcc);
318       // tests partial flush: roll on a partial flush, and ensure that wal is not archived.
319       wal.startCacheFlush(hri1.getEncodedNameAsBytes(), t1.getFamiliesKeys());
320       wal.rollWriter();
321       wal.completeCacheFlush(hri1.getEncodedNameAsBytes());
322       assertEquals(1, wal.getNumRolledLogFiles());
323     } finally {
324       if (wal != null) {
325         wal.close();
326       }
327     }
328   }

346   /**
347    * Test flush for sure has a sequence id that is beyond the last edit appended.  We do this
348    * by slowing appends in the background ring buffer thread while in foreground we call
349    * flush.  The addition of the sync over HRegion in flush should fix an issue where flush was
350    * returning before all of its appends had made it out to the WAL (HBASE-11109).
351    * see HBASE-11109
352    * @throws IOException
353    */
354   @Test
355   public void testFlushSequenceIdIsGreaterThanAllEditsInHFile() throws IOException {
356     String testName = "testFlushSequenceIdIsGreaterThanAllEditsInHFile";
357     final TableName tableName = TableName.valueOf(testName);
358     final HRegionInfo hri = new HRegionInfo(tableName);
359     final byte[] rowName = tableName.getName();
360     final HTableDescriptor htd = new HTableDescriptor(tableName);
361     htd.addFamily(new HColumnDescriptor("f"));
362     HRegion r = HRegion.createHRegion(hri, TEST_UTIL.getDefaultRootDirPath(),
363       TEST_UTIL.getConfiguration(), htd);
364     HRegion.closeHRegion(r);
365     final int countPerFamily = 10;
366     final MutableBoolean goslow = new MutableBoolean(false);
367     // subclass and doctor a method.
368     FSHLog wal = new FSHLog(FileSystem.get(conf), TEST_UTIL.getDefaultRootDirPath(),
369         testName, conf) {
370       @Override
371       void atHeadOfRingBufferEventHandlerAppend() {
372         if (goslow.isTrue()) {
373           Threads.sleep(100);
374           LOG.debug("Sleeping before appending 100ms");
375         }
376         super.atHeadOfRingBufferEventHandlerAppend();
377       }
378     };
379     HRegion region = HRegion.openHRegion(TEST_UTIL.getConfiguration(),
380       TEST_UTIL.getTestFileSystem(), TEST_UTIL.getDefaultRootDirPath(), hri, htd, wal);
381     EnvironmentEdge ee = EnvironmentEdgeManager.getDelegate();
382     try {
383       List<Put> puts = null;
384       for (HColumnDescriptor hcd: htd.getFamilies()) {
385         puts =
386           TestWALReplay.addRegionEdits(rowName, hcd.getName(), countPerFamily, ee, region, "x");
387       }
388 
389       // Now assert edits made it in.
390       final Get g = new Get(rowName);
391       Result result = region.get(g);
392       assertEquals(countPerFamily * htd.getFamilies().size(), result.size());
393 
394       // Construct a WALEdit and add it a few times to the WAL.
395       WALEdit edits = new WALEdit();
396       for (Put p: puts) {
397         CellScanner cs = p.cellScanner();
398         while (cs.advance()) {
399           edits.add(cs.current());
400         }
401       }
402       // Add any old cluster id.
403       List<UUID> clusterIds = new ArrayList<UUID>();
404       clusterIds.add(UUID.randomUUID());
405       // Now make appends run slow.
406       goslow.setValue(true);
407       for (int i = 0; i < countPerFamily; i++) {
408         final HRegionInfo info = region.getRegionInfo();
409         final WALKey logkey = new WALKey(info.getEncodedNameAsBytes(), tableName,
410             System.currentTimeMillis(), clusterIds, -1, -1, region.getMVCC());
411         wal.append(htd, info, logkey, edits, true);
412       }
413       region.flush(true);
414       // FlushResult.flushSequenceId is not visible here so go get the current sequence id.
415       long currentSequenceId = region.getSequenceId();
416       // Now release the appends
417       goslow.setValue(false);
418       synchronized (goslow) {
419         goslow.notifyAll();
420       }
421       assertTrue(currentSequenceId >= region.getSequenceId());
422     } finally {
423       region.close(true);
424       wal.close();
425     }
426   }

457   /**
458    * Test case for https://issues.apache.org/jira/browse/HBASE-16721
459    */
460   @Test (timeout = 30000)
461   public void testUnflushedSeqIdTracking() throws IOException, InterruptedException {
462     final String name = "testSyncRunnerIndexOverflow";
463     final byte[] b = Bytes.toBytes("b");
464 
465     final AtomicBoolean startHoldingForAppend = new AtomicBoolean(false);
466     final CountDownLatch holdAppend = new CountDownLatch(1);
467     final CountDownLatch flushFinished = new CountDownLatch(1);
468     final CountDownLatch putFinished = new CountDownLatch(1);
469 
470     try (FSHLog log =
471         new FSHLog(fs, FSUtils.getRootDir(conf), name, HConstants.HREGION_OLDLOGDIR_NAME, conf,
472             null, true, null, null)) {
473 
474       log.registerWALActionsListener(new WALActionsListener.Base() {
475         @Override
476         public void visitLogEntryBeforeWrite(HTableDescriptor htd, WALKey logKey, WALEdit logEdit) {
477           if (startHoldingForAppend.get()) {
478             try {
479               holdAppend.await();
480             } catch (InterruptedException e) {
481               LOG.error(e);
482             }
483           }
484         }
485       });
486 
487       // open a new region which uses this WAL
488       HTableDescriptor htd =
489           new HTableDescriptor(TableName.valueOf("t1")).addFamily(new HColumnDescriptor(b));
490       HRegionInfo hri =
491           new HRegionInfo(htd.getTableName(), HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
492 
493       final HRegion region = TEST_UTIL.createLocalHRegion(hri, htd, log);
494       ExecutorService exec = Executors.newFixedThreadPool(2);
495 
496       // do a regular write first because of memstore size calculation.
497       region.put(new Put(b).addColumn(b, b,b));
498 
499       startHoldingForAppend.set(true);
500       exec.submit(new Runnable() {
501         @Override
502         public void run() {
503           try {
504             region.put(new Put(b).addColumn(b, b,b));
505             putFinished.countDown();
506           } catch (IOException e) {
507             LOG.error(e);
508           }
509         }
510       });
511 
512       // give the put a chance to start
513       Threads.sleep(3000);
514 
515       exec.submit(new Runnable() {
516         @Override
517         public void run() {
518           try {
519             Region.FlushResult flushResult = region.flush(true);
520             LOG.info("Flush result:" +  flushResult.getResult());
521             LOG.info("Flush succeeded:" +  flushResult.isFlushSucceeded());
522             flushFinished.countDown();
523           } catch (IOException e) {
524             LOG.error(e);
525           }
526         }
527       });
528 
529       // give the flush a chance to start. Flush should have got the region lock, and
530       // should have been waiting on the mvcc complete after this.
531       Threads.sleep(3000);
532 
533       // let the append to WAL go through now that the flush already started
534       holdAppend.countDown();
535       putFinished.await();
536       flushFinished.await();
537 
538       // check whether flush went through
539       assertEquals("Region did not flush?", 1, region.getStoreFileList(new byte[][]{b}).size());
540 
541       // now check the region's unflushed seqIds.
542       long seqId = log.getEarliestMemstoreSeqNum(hri.getEncodedNameAsBytes());
543       assertEquals("Found seqId for the region which is already flushed",
544           HConstants.NO_SEQNUM, seqId);
545 
546       region.close();
547     }
548   }
