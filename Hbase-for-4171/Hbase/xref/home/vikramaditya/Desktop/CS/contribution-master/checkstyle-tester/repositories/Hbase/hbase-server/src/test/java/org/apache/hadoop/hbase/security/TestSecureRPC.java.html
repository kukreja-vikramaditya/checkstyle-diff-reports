78    static final BlockingService SERVICE =
79        TestRpcServiceProtos.TestProtobufRpcProto.newReflectiveBlockingService(

81  
82              @Override
83              public TestProtos.EmptyResponseProto ping(RpcController controller,
84                                                        TestProtos.EmptyRequestProto request)
85                  throws ServiceException {
86                return null;
87              }
88  
89              @Override
90              public TestProtos.EmptyResponseProto error(RpcController controller,
91                                                         TestProtos.EmptyRequestProto request)
92                  throws ServiceException {
93                return null;
94              }
95  
96              @Override
97              public TestProtos.EchoResponseProto echo(RpcController controller,
98                                                       TestProtos.EchoRequestProto request)
99                  throws ServiceException {
100               if (controller instanceof PayloadCarryingRpcController) {
101                 PayloadCarryingRpcController pcrc = (PayloadCarryingRpcController) controller;
102                 // If cells, scan them to check we are able to iterate what we were given and since
103                 // this is
104                 // an echo, just put them back on the controller creating a new block. Tests our
105                 // block
106                 // building.
107                 CellScanner cellScanner = pcrc.cellScanner();
108                 List<Cell> list = null;
109                 if (cellScanner != null) {
110                   list = new ArrayList<Cell>();
111                   try {
112                     while (cellScanner.advance()) {
113                       list.add(cellScanner.current());
114                     }
115                   } catch (IOException e) {
116                     throw new ServiceException(e);
117                   }
118                 }
119                 cellScanner = CellUtil.createCellScanner(list);
120                 ((PayloadCarryingRpcController) controller).setCellScanner(cellScanner);
121               }
122               return TestProtos.EchoResponseProto.newBuilder()
123                   .setMessage(request.getMessage()).build();
124             }
125           });

169   private void testRpcCallWithEnabledKerberosSaslAuth(Class<? extends RpcClient> rpcImplClass)
170       throws Exception {
171     String krbKeytab = getKeytabFileForTesting();
172     String krbPrincipal = getPrincipalForTesting();
173 
174     UserGroupInformation ugi = loginKerberosPrincipal(krbKeytab, krbPrincipal);
175     UserGroupInformation ugi2 = UserGroupInformation.getCurrentUser();
176 
177     // check that the login user is okay:
178     assertSame(ugi, ugi2);
179     assertEquals(AuthenticationMethod.KERBEROS, ugi.getAuthenticationMethod());
180     assertEquals(krbPrincipal, ugi.getUserName());
181 
182     Configuration clientConf = getSecuredConfiguration();
183     callRpcService(rpcImplClass, User.create(ugi2), clientConf, false);
184   }
185 
186   private UserGroupInformation loginKerberosPrincipal(String krbKeytab, String krbPrincipal)
187       throws Exception {
188     Configuration cnf = new Configuration();
189     cnf.set(CommonConfigurationKeys.HADOOP_SECURITY_AUTHENTICATION, "kerberos");
190     UserGroupInformation.setConfiguration(cnf);
191     UserGroupInformation.loginUserFromKeytab(krbPrincipal, krbKeytab);
192     return UserGroupInformation.getLoginUser();
193   }

195   private void callRpcService(Class<? extends RpcClient> rpcImplClass, User clientUser,
196                               Configuration clientConf, boolean allowInsecureFallback)
197       throws Exception {
198     Configuration clientConfCopy = new Configuration(clientConf);
199     clientConfCopy.set(RpcClientFactory.CUSTOM_RPC_CLIENT_IMPL_CONF_KEY, rpcImplClass.getName());
200 
201     Configuration conf = getSecuredConfiguration();
202     conf.setBoolean(RpcServer.FALLBACK_TO_INSECURE_CLIENT_AUTH, allowInsecureFallback);
203 
204     SecurityInfo securityInfoMock = Mockito.mock(SecurityInfo.class);
205     Mockito.when(securityInfoMock.getServerPrincipal())
206         .thenReturn(HBaseKerberosUtils.KRB_PRINCIPAL);
207     SecurityInfo.addInfo("TestProtobufRpcProto", securityInfoMock);
208 
209     InetSocketAddress isa = new InetSocketAddress(HOST, 0);
210 
211     RpcServerInterface rpcServer =
212         new RpcServer(null, "AbstractTestSecureIPC",
213             Lists.newArrayList(new RpcServer.BlockingServiceAndInterface(SERVICE, null)), isa,
214             conf, new FifoRpcScheduler(conf, 1));
215     rpcServer.start();
216     try (RpcClient rpcClient = RpcClientFactory.createClient(clientConf,
217         HConstants.DEFAULT_CLUSTER_ID.toString())) {
218       InetSocketAddress address = rpcServer.getListenerAddress();
219       if (address == null) {
220         throw new IOException("Listener channel is closed");
221       }
222       BlockingRpcChannel channel =
223           rpcClient.createBlockingRpcChannel(
224 
225             ServerName.valueOf(address.getHostName(), address.getPort(),
226             System.currentTimeMillis()), clientUser, 5000);
227       TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface stub =
228           TestRpcServiceProtos.TestProtobufRpcProto.newBlockingStub(channel);
229       List<String> results = new ArrayList<String>();
230       TestThread th1 = new TestThread(stub, results);
231       th1.start();
232       th1.join();
233 
234     } finally {
235       rpcServer.stop();
236     }
237   }

239   public void testRpcFallbackToSimpleAuth(Class<? extends RpcClient> rpcImplClass) throws Exception {
240     String krbKeytab = getKeytabFileForTesting();
241     String krbPrincipal = getPrincipalForTesting();
242 
243     UserGroupInformation ugi = loginKerberosPrincipal(krbKeytab, krbPrincipal);
244     assertEquals(AuthenticationMethod.KERBEROS, ugi.getAuthenticationMethod());
245     assertEquals(krbPrincipal, ugi.getUserName());
246 
247     String clientUsername = "testuser";
248     UserGroupInformation clientUgi = UserGroupInformation.createUserForTesting(clientUsername,
249         new String[]{clientUsername});
250 
251     // check that the client user is insecure
252     assertNotSame(ugi, clientUgi);
253     assertEquals(AuthenticationMethod.SIMPLE, clientUgi.getAuthenticationMethod());
254     assertEquals(clientUsername, clientUgi.getUserName());
255 
256     Configuration clientConf = new Configuration();
257     clientConf.set(User.HBASE_SECURITY_CONF_KEY, "simple");
258     callRpcService(rpcImplClass, User.create(clientUgi), clientConf, true);
259   }

261   public static class TestThread extends Thread {
262       private final TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface stub;
263 
264       private final List<String> results;
265 
266           public TestThread(TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface stub, List<String> results) {
267           this.stub = stub;
268           this.results = results;
269         }
270 
271           @Override
272       public void run() {
273           String result;
274           try {
275               result = stub.echo(null, TestProtos.EchoRequestProto.newBuilder().setMessage(String.valueOf(
276                   ThreadLocalRandom.current().nextInt())).build()).getMessage();
277             } catch (ServiceException e) {
278               throw new RuntimeException(e);
279             }
280           if (results != null) {
281               synchronized (results) {
282                   results.add(result);
283                 }
284             }
285         }
286     }
