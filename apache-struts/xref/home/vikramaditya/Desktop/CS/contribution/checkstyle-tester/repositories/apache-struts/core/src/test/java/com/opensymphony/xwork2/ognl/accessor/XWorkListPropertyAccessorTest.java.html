12  import java.util.List;

14  /**
15   * @author Jason Carreira <jcarreira@eplus.com>
16   */
17  public class XWorkListPropertyAccessorTest extends XWorkTestCase {
18  
19      public void testContains() {
20          ValueStack vs = ActionContext.getContext().getValueStack();
21          ListHolder listHolder = new ListHolder();
22          vs.push(listHolder);
23  
24          vs.setValue("longs", new String[] {"1", "2", "3"});
25  
26          assertNotNull(listHolder.getLongs());
27          assertEquals(3, listHolder.getLongs().size());
28          assertEquals(new Long(1), (Long) listHolder.getLongs().get(0));
29          assertEquals(new Long(2), (Long) listHolder.getLongs().get(1));
30          assertEquals(new Long(3), (Long) listHolder.getLongs().get(2));
31  
32          assertTrue(((Boolean) vs.findValue("longs.contains(1)")).booleanValue());
33      }
34  
35      public void testCanAccessListSizeProperty() {
36          ValueStack vs = ActionContext.getContext().getValueStack();
37          List myList = new ArrayList();
38          myList.add("a");
39          myList.add("b");
40  
41          ListHolder listHolder = new ListHolder();
42          listHolder.setStrings(myList);
43  
44          vs.push(listHolder);
45  
46          assertEquals(new Integer(myList.size()), vs.findValue("strings.size()"));
47          assertEquals(new Integer(myList.size()), vs.findValue("strings.size"));
48      }
49  }
